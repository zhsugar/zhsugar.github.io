#define LED 2
			#define shengyin 3
			int flag=0;//定义一个变量记录小灯是亮起还是熄灭
			const int ledRedPin = 13;
			const int ledGreenPin = 12;
			const int analogPin = A0;
			const int digitalPin = 7;
			const int sound = 11;
			int Astate = 0;
			boolean  Dstate = 0;
			void setup()
			{
			  pinMode(LED,OUTPUT);//定义小灯为输出模式
			  pinMode(shengyin,INPUT);//定义声音控制口为输入模式
			
			  pinMode (digitalPin, INPUT);
			  pinMode(ledRedPin, OUTPUT);
			  pinMode(ledGreenPin, OUTPUT);
			  pinMode(sound, OUTPUT);
			  Serial.begin(9600);
			}
			void loop()
			{
			  Astate = analogRead(analogPin);
			  Serial.println(Astate);
			  Dstate = digitalRead(digitalPin);
			  Serial.println(Dstate);
			  if ( Dstate == HIGH )
			  {
			    digitalWrite(ledRedPin, LOW);
			    digitalWrite(ledGreenPin, HIGH);
			    //停止发声
			    noTone(sound);
			    Serial.println("蜂鸣器停止鸣叫");
			  }
			  if ( Dstate == LOW)
			  {
			
			    digitalWrite(ledGreenPin, LOW);
			    digitalWrite(ledRedPin, HIGH);
			    for (int i = 200; i <= 210; i++)
			    {
			      tone(sound, i);
			      Serial.println("蜂鸣器开始鸣叫");
			      //delay(5);
			    }
			  }
			  //delay(200);
			}
