#define PIN_DO 2//数字口2   这个是接土壤湿度传感器数字信号的D0
			int relayPin=9;//定义数字2接口接继电器的控制口
			
			void setup()
			{
			  pinMode(relayPin,OUTPUT);//定义继电器接口为输出接口
			  pinMode(PIN_DO, INPUT);//对应的口设置为输入模式  
			  Serial.begin(9600);  //设置串口波特率
			}
			void loop()
			{
			  
			  Serial.print("数字量值为：");//字符串显示  
			  Serial.println(digitalRead(PIN_DO));//串口输出对应的数字量
			
			  if(digitalRead(PIN_DO)){
			    
			     digitalWrite(relayPin,LOW);//开始浇水
			     Serial.print("开始浇水");//字符串显示 
			  }else{
			    
			    digitalWrite(relayPin,HIGH);//断开浇水
			    Serial.print("停止浇水");//字符串显示 
			  }
			  
			 
			  delay(500);//延时
			  
			 }
